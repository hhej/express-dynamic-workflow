    ```python
    """TOOL-05: Tavily-backed fuel news / market trends search.

    D-09 trigger semantics: planner routes to search_context ONLY for
    news/trend questions (e.g. "why is fuel up", "diesel news this week").
    Standard surcharge_query and followup_query NEVER trigger search.

    D-12 graceful failure: any Tavily error (network, quota, auth) raises
    a RuntimeError that search_agent_node converts to a warn-status trace
    entry; search_context stays None; planner continues unblocked.

    D-12 caching: TTLCache (default 30 min) keyed on normalized query.
    """
    from __future__ import annotations

    import logging
    import os
    import re
    from datetime import datetime, timezone
    from typing import Tuple

    from backend.agent.tools._cache import TTLCache
    from backend.agent.tools.models import SearchResult, SearchSource
    from backend.config import SEARCH_CACHE_TTL_SECONDS

    __all__ = ["search_fuel_news", "_normalize_query", "_clear_cache"]

    logger = logging.getLogger(__name__)

    # In-process cache shared across all callers — emptied via _clear_cache() in tests.
    _CACHE: TTLCache = TTLCache(ttl_seconds=SEARCH_CACHE_TTL_SECONDS)

    _WS_RE = re.compile(r"\s+")


    def _normalize_query(query: str) -> str:
        """Pitfall 3 mitigation: lowercase + strip + collapse multi-space.

        Ensures semantically identical queries hash to the same cache key.
        """
        return _WS_RE.sub(" ", query.strip()).lower()


    def _clear_cache() -> None:
        """Test-only — reset the module-level cache."""
        _CACHE.clear()


    def _client():
        """Lazy Tavily client construction so tests can monkeypatch tavily.TavilyClient."""
        api_key = os.environ.get("TAVILY_API_KEY", "")
        if not api_key:
            raise RuntimeError(
                "TAVILY_API_KEY is empty — search_fuel_news cannot run"
            )
        from tavily import TavilyClient  # type: ignore[import-untyped]
        return TavilyClient(api_key)


    def search_fuel_news(query: str, max_results: int = 5) -> SearchResult:
        """Tavily news search for fuel market context.

        Args:
            query: User-supplied question. Will be normalized for caching.
            max_results: Hard ceiling on returned sources (1..10).

        Returns:
            SearchResult with optional 1-line summary + ranked sources.

        Raises:
            RuntimeError: TAVILY_API_KEY missing or any Tavily call failure.
        """
        norm = _normalize_query(query)
        key: Tuple[str, int] = (norm, max_results)
        cached = _CACHE.get(key)
        if cached is not None:
            return cached

        try:
            raw = _client().search(
                query=query,
                topic="news",
                max_results=max_results,
                include_answer="basic",
                search_depth="basic",
            )
        except RuntimeError:
            raise
        except Exception as exc:  # noqa: BLE001 — convert ANY Tavily failure
            logger.warning("Tavily search failed: %s", exc)
            raise RuntimeError(f"Tavily search failed: {exc}") from exc

        sources = []
        for r in (raw.get("results") or []):
            content = (r.get("content") or "")[:240]
            sources.append(SearchSource(
                title=r.get("title", ""),
                url=r.get("url", ""),
                snippet=content,
                published_at=r.get("published_date"),
            ))
        result = SearchResult(
            query=query,
            summary=raw.get("answer"),
            sources=sources,
            fetched_at=datetime.now(timezone.utc)
                .isoformat().replace("+00:00", "Z"),
        )
        _CACHE.set(key, result)
        return result
    ```

    Step 2 — Create `backend/agent/nodes/search_agent.py` (NEW file). Mirror fuel_agent_node structure verbatim:
    ```python
    """ORCH-style Search Agent node (TOOL-05).

    Mirrors fuel_agent / route_agent narration pattern:
    - Calls search_fuel_news tool.
    - Narrates via Gemini with D-11 deterministic fallback.
    - Emits ONE D-12-shape reasoning_trace entry.
    - Populates state.search_context (D-11 shape).

    D-12 graceful failure: any RuntimeError from the tool → search_context=None,
    trace status="warn", planner continues. Search NEVER blocks the surcharge
    response.
    """
    from __future__ import annotations

    import json
    import logging
    from datetime import datetime, timezone
    from typing import Optional

    from langchain_core.messages import HumanMessage, SystemMessage
    from pydantic import BaseModel, Field, ValidationError

    from backend.agent.llm import get_chat_model
    from backend.agent.prompts.search_agent import SYSTEM_PROMPT
    from backend.agent.tools.search_fuel_news import search_fuel_news
    from backend.agent.tools.models import SearchResult

    __all__ = ["search_agent_node"]

    logger = logging.getLogger(__name__)

    _DEFAULT_QUERY = "Thailand diesel fuel price news"


    class SearchReasoning(BaseModel):
        """Structured narration schema (D-11)."""
        summary: str = Field(description="1-2 sentence market context")


    def _last_user_message(state: dict) -> str:
        for m in reversed(state.get("messages") or []):
            if m.get("role") == "user":
                return m.get("content") or ""
        return ""


    def _deterministic_narration(result: SearchResult) -> str:
        """D-11 fallback when LLM narration fails."""
        if result.summary:
            return result.summary
        if result.sources:
            top = result.sources[0]
            snippet = top.snippet[:160] if top.snippet else top.title
            return snippet
        return "No recent news found about Thailand diesel prices."


    def _parse_structured(raw: str) -> SearchReasoning:
        text = raw.strip()
        if text.startswith("```"):
            lines = text.splitlines()[1:]
            if lines and lines[-1].strip().startswith("```"):
                lines = lines[:-1]
            text = "\n".join(lines).strip()
        return SearchReasoning.model_validate(json.loads(text))


    def _narrate_with_llm(result: SearchResult) -> str:
        try:
            model = get_chat_model()
            user_payload = json.dumps({
                "query": result.query,
                "tavily_summary": result.summary,
                "sources": [
                    {"title": s.title, "snippet": s.snippet[:160]}
                    for s in result.sources[:3]
                ],
            }, ensure_ascii=False)
            response = model.invoke([
                SystemMessage(content=SYSTEM_PROMPT),
                HumanMessage(content=user_payload),
            ])
            content = getattr(response, "content", response)
            if not isinstance(content, str):
                content = str(content)
            parsed = _parse_structured(content)
            return parsed.summary
        except (Exception, ValidationError) as exc:
            logger.warning("search_agent narration fallback: %s", exc)
            return _deterministic_narration(result)


    def search_agent_node(state: dict) -> dict:
        """Search → narrate → populate search_context + emit one trace entry."""
        query = _last_user_message(state) or _DEFAULT_QUERY
        prior = len(state.get("reasoning_trace") or [])
        timestamp = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")

        try:
            result = search_fuel_news(query=query, max_results=5)
        except RuntimeError as exc:
            logger.warning("search_fuel_news failed: %s", exc)
            return {
                "search_context": None,
                "reasoning_trace": [{
                    "step": prior + 1,
                    "agent": "search_agent",
                    "tool": "search_fuel_news",
                    "tool_input": {"query": query},
                    "tool_output": {"error": str(exc)},
                    "reasoning": "Search failed; continuing without market context.",
                    "timestamp": timestamp,
                    "status": "warn",
                }],
            }

        narration = _narrate_with_llm(result)

        return {
            "search_context": result.model_dump(),
            "reasoning_trace": [{
                "step": prior + 1,
