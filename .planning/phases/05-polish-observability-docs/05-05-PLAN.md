Implement ORCH-09 HITL approval gate. When `surcharge_result.total > HITL_TOTAL_THB_THRESHOLD` the graph pauses via LangGraph's native `interrupt()` primitive between `pricing_agent` and `response_node`. The chat handler emits a sixth SSE event `approval_required` with the surcharge breakdown, and the frontend (Plan 05-06) renders Approve/Deny buttons. Resume happens via a follow-up `POST /api/chat` with `{thread_id, approve: bool}` that calls `graph.ainvoke(Command(resume=approve), config)`. On approve, response_node renders the standard status='ok' answer with the breakdown table. On deny, response_node emits status='partial' with a "user declined" prose and NO breakdown table (D-07).

    ```python
    """ORCH-09: HITL approval gate (D-04 / D-05 / D-08).

    Placed between pricing_agent and response_node. For low-value totals
    (<= HITL_TOTAL_THB_THRESHOLD) the gate is a pass-through that sets
    approval_decision='approve' with NO trace entry — zero overhead on the
    common case. For high-value totals the gate emits a 'warn' trace entry
    AND calls langgraph.types.interrupt(), suspending the run until the
    chat handler resumes via Command(resume=approve_bool).

    The interrupt() return value is the resume value supplied by the
    caller; we map True → 'approve', anything else → 'deny' and emit a
    second 'ok' trace entry recording the decision (D-08).
    """
    from __future__ import annotations

    import logging
    from datetime import datetime, timezone

    from langgraph.types import interrupt

    from backend.config import HITL_TOTAL_THB_THRESHOLD

    __all__ = ["hitl_gate_node"]

    logger = logging.getLogger(__name__)


    def hitl_gate_node(state: dict) -> dict:
        """D-05 gate. Pass-through for low-value, interrupt() for high-value."""
        sr = state.get("surcharge_result") or {}
        total = float(sr.get("total") or 0.0)

        if total <= HITL_TOTAL_THB_THRESHOLD:
            # Low-value pass-through. Zero trace entries, just sets the
            # decision so response_node sees approval_decision='approve'.
            return {"approval_decision": "approve"}

        prior = len(state.get("reasoning_trace") or [])
        ts = lambda: datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")

        pre_trace = {
            "step": prior + 1,
            "agent": "hitl_gate",
            "tool": "interrupt",
            "tool_input": {
                "threshold": HITL_TOTAL_THB_THRESHOLD,
                "total": total,
            },
            "tool_output": {"decision_pending": True},
            "reasoning": (
                f"Total {total:.2f} THB exceeds threshold "
                f"{HITL_TOTAL_THB_THRESHOLD:.2f} THB; awaiting user approval."
            ),
            "timestamp": ts(),
            "status": "warn",
        }

        # interrupt() returns the resume value when the graph is invoked
        # again with Command(resume=value). Caller passes a bool.
        decision_value = interrupt({
            "type": "approval_required",
            "surcharge_result": sr,
            "threshold": HITL_TOTAL_THB_THRESHOLD,
        })

        approval = "approve" if decision_value in (True, "approve") else "deny"

        post_trace = {
            "step": prior + 2,
            "agent": "hitl_gate",
            "tool": "interrupt",
            "tool_input": {},
            "tool_output": {"decision": approval},
            "reasoning": f"User decision: {approval}",
            "timestamp": ts(),
            "status": "ok",
        }

        return {
            "approval_decision": approval,
            "reasoning_trace": [pre_trace, post_trace],
        }
    ```

    ```python
            pending_approval = False
            try:
                initial_state = {
                    "messages": [{"role": "user", "content": message}],
                }
                async for kind, payload in _drain_events(graph, initial_state, config):
                    yield format_sse(kind, payload)
                # After stream ends: check for interrupt.
                snapshot = await graph.aget_state(config)
                if snapshot.next:
                    pending_approval = True
                    iv = (
                        snapshot.tasks[0].interrupts[0].value
                        if snapshot.tasks and snapshot.tasks[0].interrupts
                        else {}
                    )
                    yield format_sse("approval_required", {
                        "thread_id": thread_id,
                        "surcharge_result": iv.get("surcharge_result"),
                        "threshold": iv.get("threshold"),
                    })
                    return  # Pitfall 2: NO done after approval_required.
            except Exception as exc:  # noqa: BLE001
                logger.exception("Chat stream failed")
                yield format_sse("error", {"message": str(exc), "retryable": False})
            finally:
                if not pending_approval:
                    yield format_sse("done", {})

        return StreamingResponse(
            gen(),
            media_type="text/event-stream",
            headers={
                "Cache-Control": "no-cache",
                "X-Accel-Buffering": "no",
                "Connection": "keep-alive",
            },
        )


    async def _resume_stream(graph, thread_id: str, approve: bool):
        """D-06 resume — Command(resume=approve)."""
        # Pitfall 1: REUSE _make_config so Langfuse callbacks + metadata
        # are preserved across the pause. turn_idx for resume is the same
        # as the paused turn (count user messages on the snapshot before resume).
        turn_idx = await _next_turn_idx(graph, thread_id)
        # _next_turn_idx returns count of user messages; resume turn doesn't
        # add a new user message, so the index is the same as the in-flight
        # turn. (-1 to point at the paused turn rather than the next one.)
        cfg_turn = max(0, turn_idx - 1) if turn_idx > 0 else 0
        config = _make_config(thread_id, cfg_turn)

        async def gen():
            yield format_sse("meta", {"thread_id": thread_id})
            try:
                async for kind, payload in _drain_events(
                    graph, Command(resume=approve), config
                ):
                    yield format_sse(kind, payload)
            except Exception as exc:  # noqa: BLE001
                logger.exception("Chat resume stream failed")
                yield format_sse("error", {"message": str(exc), "retryable": False})
            finally:
                yield format_sse("done", {})

        return StreamingResponse(
            gen(),
            media_type="text/event-stream",
            headers={
                "Cache-Control": "no-cache",
                "X-Accel-Buffering": "no",
                "Connection": "keep-alive",
            },
        )
    ```
